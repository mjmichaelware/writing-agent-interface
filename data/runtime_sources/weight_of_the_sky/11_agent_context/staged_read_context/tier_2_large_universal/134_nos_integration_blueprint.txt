STEP 2.1 — STAR WARS CONTINUOUS SCROLL FADE (Project Checkpoint 2 §2.1)

  Current: IntersectionObserver with discrete active/inactive paragraph states.
  Target:  Every paragraph computes opacity and blur per-frame based on the
           distance from its center to the viewport focal line. Paragraphs
           crossing the focal line are crisp (blur 0, opacity 1). Paragraphs
           halfway out are blurred and faded. Paragraphs past the viewport
           edge are blurred to 8px and faded to 0 opacity.

  Implementation file: src/components/layers/canvas/ManuscriptCore.tsx

  Replace the IntersectionObserver with:
    - A ref to the scroll container (window if scroll source is window).
    - A requestAnimationFrame loop on scroll events (passive listener).
    - For each paragraph, compute the bounding rect, then:
        const elementCenter = rect.top + rect.height / 2
        const viewportCenter = window.innerHeight / 2
        const distance = Math.abs(elementCenter - viewportCenter)
        const focalRange = window.innerHeight * 0.5
        const ratio = Math.min(1, distance / focalRange)
        el.style.opacity = String(1 - ratio * 0.7)
        el.style.filter = `blur(${ratio * 3}px)`
        el.style.transform = `translateY(${ratio * 8}px) scale(${1 - ratio * 0.03})`
    - The active paragraph (closest to viewport center) emits scroll:focus
      with paraIndex, content, and slug — same payload as currently.

  This is GPU-friendly because we write style directly, no React state.

IMPLEMENTATION

  Modify src/app/api/agent/route.ts to detect multimodal payloads and
  route accordingly:

    import { NextResponse } from "next/server";

    export async function POST(request: Request) {
      try {
        const body = await request.json();
        const prompt = String(body?.prompt || body?.query || "").trim();
        const context = String(body?.context || "").trim();
        const imageData = body?.imageData;  // base64 or URL
        const mode = body?.mode || "auto";  // "claude" | "gemini" | "auto"

        if (!prompt) {
          return NextResponse.json({ error: "Missing prompt" }, { status: 400 });
        }

        // Routing decision per AIM Blueprint architecture
        const useGemini =
          mode === "gemini" ||
          (mode === "auto" && imageData);  // multimodal → Gemini

        if (useGemini) {
          return await queryGemini(prompt, context, imageData);
        }

        return await queryOpenAI(prompt, context);
      } catch (err) {
        return NextResponse.json(
          { error: err instanceof Error ? err.message : "Agent failure" },
          { status: 500 }
        );
      }
    }

    async function queryGemini(prompt: string, context: string, imageData?: any) {
      if (!process.env.GEMINI_API_KEY) {
        return NextResponse.json({
          response: "Gemini multimodal offline: missing GEMINI_API_KEY.",
        });
      }

      const parts: any[] = [
        { text: `Context:\n${context}\n\nPrompt:\n${prompt}` }
      ];

      if (imageData) {
        parts.push({
          inline_data: {
            mime_type: "image/jpeg",
            data: imageData.replace(/^data:image\/[a-z]+;base64,/, ""),
          }
        });
      }

      const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash-exp:generateContent?key=${process.env.GEMINI_API_KEY}`;

      const res = await fetch(url, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          contents: [{ role: "user", parts }],
          systemInstruction: {
            parts: [{
              text: "You are the multimodal vision layer of the Writing Agent " +
                    "for The Weight of the Sky. Analyze imagery, diagrams, and " +
                    "screenshots in the context of narrative motifs, dualisms, " +
                    "and biblical references. Never reveal infrastructure."
            }]
          },
          generationConfig: { temperature: 0.7, maxOutputTokens: 2048 },
        }),
      });

      if (!res.ok) {
        return NextResponse.json({ response: "Gemini request failed." });
      }

      const data = await res.json();
      const output = data?.candidates?.[0]?.content?.parts?.[0]?.text || "";
      return NextResponse.json({ response: output, provider: "gemini" });
    }

    async function queryOpenAI(prompt: string, context: string) {
      if (!process.env.OPENAI_API_KEY) {
        return NextResponse.json({
          response: "OpenAI offline: missing OPENAI_API_KEY.",
        });
      }

      const response = await fetch("https://api.openai.com/v1/chat/completions", {
        method: "POST",
        headers: {
          Authorization: `Bearer ${process.env.OPENAI_API_KEY}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          model: "gpt-4o",
          temperature: 0.7,
          messages: [
            {
              role: "system",
              content:
                "You are the Writing Agent for The Weight of the Sky by " +
                "Michael Alonza Prentice Ware. Analyze motifs, dualisms, " +
                "Jungian archetypes, biblical references, character arcs, " +
                "and narrative architecture. Never reveal infrastructure."
            },
            {
              role: "user",
              content: context ? `Context:\n${context}\n\nPrompt:\n${prompt}` : prompt
            }
          ],
        }),
      });

      if (!response.ok) {
        return NextResponse.json({ response: "OpenAI request failed." });
      }

      const data = await response.json();
      const output = data?.choices?.[0]?.message?.content || "No response generated.";
      return NextResponse.json({ response: output, provider: "openai" });
    }

  Create src/services/document-analyzer/gdrive-sync.ts:

    import { google } from "googleapis";
    import { promises as fs } from "fs";
    import path from "path";

    const auth = new google.auth.OAuth2(
      process.env.GOOGLE_CLIENT_ID,
      process.env.GOOGLE_CLIENT_SECRET,
      process.env.GOOGLE_REDIRECT_URI
    );

    auth.setCredentials({
      refresh_token: process.env.GOOGLE_REFRESH_TOKEN,
    });

    const drive = google.drive({ version: "v3", auth });

    const FOLDER_ID = process.env.GDRIVE_MANUSCRIPT_FOLDER_ID!;
    const LOCAL_DIR = path.join(
      process.cwd(),
      "src/data-layer/ingestion-buffer/gdrive_raw"
    );

    export async function syncDriveFolder(): Promise<{
      synced: number;
      errors: string[];
    }> {
      const errors: string[] = [];
      let synced = 0;

      const res = await drive.files.list({
        q: `'${FOLDER_ID}' in parents and mimeType='text/plain' and trashed=false`,
        fields: "files(id, name, modifiedTime)",
        pageSize: 1000,
      });

      const files = res.data.files || [];
      await fs.mkdir(LOCAL_DIR, { recursive: true });

      for (const file of files) {
        try {
          const exported = await drive.files.get({
            fileId: file.id!,
            alt: "media",
          });
          const localPath = path.join(LOCAL_DIR, file.name!);
          await fs.writeFile(localPath, String(exported.data));
          synced++;
        } catch (err) {
          errors.push(`${file.name}: ${err}`);
        }
      }

      return { synced, errors };
    }

  Create src/app/api/sync/drive/route.ts:

    import { NextResponse } from "next/server";

      }
    }

    async storeEmbedding(
      paragraphId: string,
      embedding: number[]
    ): Promise<void> {
      const vectorString = `[${embedding.join(",")}]`;
      await this.pool.query(
        `UPDATE paragraphs SET embedding = $1::vector WHERE id = $2`,
        [vectorString, paragraphId]
      );
    }

    async semanticSearch(
      queryEmbedding: number[],
      limit: number = 20
    ): Promise<{
      ref: string;
      title: string;
      note: string;
      content: string;
      para_index: number;
      similarity: number;
    }[]> {
      const vectorString = `[${queryEmbedding.join(",")}]`;
      const res = await this.pool.query(
        `SELECT p.content, p.chunk_index, c.manifest_id,
                1 - (p.embedding <=> $1::vector) AS similarity
         FROM paragraphs p
         JOIN chapters c ON p.chapter_id = c.id
         WHERE p.embedding IS NOT NULL
         ORDER BY p.embedding <=> $1::vector
         LIMIT $2`,
        [vectorString, limit]
      );
      return res.rows.map((r) => ({
        ref: String(r.manifest_id || "Unknown"),
        title: String(r.content || "").slice(0, 64),
        note: String(r.content || ""),
        content: String(r.content || ""),
        para_index: Number(r.chunk_index || 0),
        similarity: Number(r.similarity || 0),
      }));
    }

    import { embedText } from "@/services/memory-engine/vertex-embedder";

    async function main() {

    export async function POST(request: Request) {
      try {
        const body = await request.json();
        const text = String(body?.text || "").trim();
        const voice = String(body?.voice || "en-US-Wavenet-D");

      });

        unsubFocus();
        unsubToggle();
        if (currentAudio) currentAudio.pause();
      };

PHASE 3.6 — IMAGEN API FOR DYNAMIC CINEMA ASSET GENERATION
================================================================================

PURPOSE
  Replace the static `bg.png`, `flies.jpg`, `megiddo1.jpg`, `megiddo2.jpg`
  with on-demand generation tied to paragraph thematic content. When a
  paragraph crosses the focal line, if the embedding doesn't match any
  cached asset within a similarity threshold, Imagen generates a new one
  in the prestige cinema aesthetic.

PREREQUISITES
  - Enable Vertex AI Imagen access.
  - Service account with Vertex AI User role.

IMPLEMENTATION

  Create src/services/image-engine/imagen-generator.ts:

    import { VertexAI } from "@google-cloud/vertexai";
    import { promises as fs } from "fs";
    import path from "path";
    import crypto from "crypto";

    const PROJECT = process.env.GOOGLE_CLOUD_PROJECT!;
    const LOCATION = "us-central1";
    const MODEL = "imagen-3.0-generate-001";

    const ASSET_DIR = path.join(process.cwd(), "public/assets/generated");
    const CACHE_INDEX = path.join(ASSET_DIR, "_index.json");

    const STYLE_PROMPT =
      "Oil painting in the style of 19th century romantic landscape art, " +
      "muted warm parchment palette, dim candle-lit interior, deep shadows, " +
      "no text, no people unless specified, biblical antiquity setting, " +
      "1003 BCE Hebron region, Mediterranean Levant, EB Garamond aesthetic.";

    export async function generateCinemaAsset(content: string): Promise<string> {
      await fs.mkdir(ASSET_DIR, { recursive: true });
      const hash = crypto.createHash("sha256")
        .update(content.slice(0, 500))
        .digest("hex")
        .slice(0, 16);
      const filename = `cinema-${hash}.png`;
      const localPath = path.join(ASSET_DIR, filename);
      const publicPath = `/assets/generated/${filename}`;

      try {
        await fs.access(localPath);
        return publicPath;
      } catch {
        // Not cached, generate
      }

      const url = `https://${LOCATION}-aiplatform.googleapis.com/v1/projects/${PROJECT}/locations/${LOCATION}/publishers/google/models/${MODEL}:predict`;

      const promptText = `${STYLE_PROMPT}\n\nScene: ${content.slice(0, 400)}`;
      const res = await fetch(url, {
        method: "POST",
        headers: {
          Authorization: `Bearer ${await getAccessToken()}`,
          "Content-Type": "application/json",
        },
        body: JSON.stringify({
          instances: [{ prompt: promptText }],
          parameters: { sampleCount: 1, aspectRatio: "9:16" },
        }),
      });

      const data = await res.json();
      const base64 = data?.predictions?.[0]?.bytesBase64Encoded;
      if (!base64) throw new Error("Imagen returned no image");

      await fs.writeFile(localPath, Buffer.from(base64, "base64"));
      return publicPath;
    }

    async function getAccessToken(): Promise<string> {
      // Use google-auth-library to fetch token via service account
      const { GoogleAuth } = await import("google-auth-library");
      const auth = new GoogleAuth({
        scopes: ["https://www.googleapis.com/auth/cloud-platform"],
      });
      const client = await auth.getClient();
      const tokenResponse = await client.getAccessToken();
      return tokenResponse.token!;
    }

    const groq = new Groq({ apiKey: process.env.GROQ_API_KEY });

    export type ThematicScore = {
      mass: number;      // 0-1, how much the prose feels heavy/descent
      tension: number;   // 0-1, how compressed or strained
      drift: number;     // 0-1, lateral instability
      blur: number;      // 0-1, dreamlike unfocused
      tone: "sacred" | "descent" | "neutral";

    };

    export async function scoreThematics(content: string): Promise<ThematicScore> {
      const response = await groq.chat.completions.create({
        model: "llama-3.3-70b-versatile",
        messages: [
          {
            role: "system",
            content:
              "Score this prose passage from The Weight of the Sky by Michael " +
              "Alonza Prentice Ware. Return ONLY valid JSON with keys: mass, " +
              "tension, drift, blur (each 0.0-1.0), tone (sacred/descent/neutral), " +
              "archetype (shadow/persona/anima/self/hero). No prose explanation."
          },
          { role: "user", content }
        ],
        response_format: { type: "json_object" },
        temperature: 0.3,
      });

      const text = response.choices[0]?.message?.content || "{}";
      return JSON.parse(text);
    }
