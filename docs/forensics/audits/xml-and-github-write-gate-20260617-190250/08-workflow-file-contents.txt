---- .github/workflows/semantic-rehash-dry-run.yml ----
name: Semantic Rehash Dry Run

on:
  workflow_dispatch:
    inputs:
      chapter:
        description: "Optional chapter number"
        required: false
      limit:
        description: "Paragraph limit"
        required: false
        default: "12"

jobs:
  dry-run:
    runs-on: ubuntu-latest
    env:
      SUPABASE_ACCESS_TOKEN: ${{ secrets.SUPABASE_ACCESS_TOKEN }}
      SUPABASE_PROJECT_REF: ${{ secrets.SUPABASE_PROJECT_REF || vars.SUPABASE_PROJECT_REF }}
      VERTEX_PROJECT_ID: ${{ vars.VERTEX_PROJECT_ID || 'ai-job-agent-498702' }}
      VERTEX_LOCATION: ${{ vars.VERTEX_LOCATION || 'us-central1' }}
      VERTEX_MODEL: ${{ vars.VERTEX_MODEL || 'gemini-2.5-pro' }}
      GOOGLE_API_KEY: ${{ secrets.GOOGLE_API_KEY }}
      GCLOUD_ACCESS_TOKEN: ${{ secrets.GCLOUD_ACCESS_TOKEN }}
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: "22"

      - name: Verify fixed source hashes
        run: node scripts/semantic/verify-context-and-xml-hashes.mjs

      - name: Dry-run XML-grounded semantic rehash
        run: |
          ARGS="--limit=${{ github.event.inputs.limit }}"
          if [ -n "${{ github.event.inputs.chapter }}" ]; then
            ARGS="$ARGS --chapter=${{ github.event.inputs.chapter }}"
          fi
          node scripts/semantic/xml-grounded-vertex-rehash.mjs $ARGS

      - name: Upload dry-run forensic output
        uses: actions/upload-artifact@v4
        with:
          name: semantic-rehash-dry-run
          path: docs/forensics/audits/xml-grounded-vertex-rehash-runs/
---- .github/workflows/semantic-rehash-full.yml ----
name: Semantic Rehash Full

on:
  workflow_dispatch:
    inputs:
      confirm:
        description: "Type REHASH to run full write-mode rehash"
        required: true
      limit:
        description: "Optional paragraph limit. Leave blank for full corpus."
        required: false

jobs:
  full-rehash:
    if: github.event.inputs.confirm == 'REHASH'
    runs-on: ubuntu-latest
    timeout-minutes: 360
    env:
      SUPABASE_ACCESS_TOKEN: ${{ secrets.SUPABASE_ACCESS_TOKEN }}
      SUPABASE_PROJECT_REF: ${{ secrets.SUPABASE_PROJECT_REF || vars.SUPABASE_PROJECT_REF }}
      VERTEX_PROJECT_ID: ${{ vars.VERTEX_PROJECT_ID || 'ai-job-agent-498702' }}
      VERTEX_LOCATION: ${{ vars.VERTEX_LOCATION || 'us-central1' }}
      VERTEX_MODEL: ${{ vars.VERTEX_MODEL || 'gemini-2.5-pro' }}
      GOOGLE_API_KEY: ${{ secrets.GOOGLE_API_KEY }}
      GCLOUD_ACCESS_TOKEN: ${{ secrets.GCLOUD_ACCESS_TOKEN }}
    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: "22"

      - name: Verify fixed source hashes
        run: node scripts/semantic/verify-context-and-xml-hashes.mjs

      - name: Full XML-grounded semantic rehash
        run: |
          ARGS="--write --register-sources"
          if [ -n "${{ github.event.inputs.limit }}" ]; then
            ARGS="$ARGS --limit=${{ github.event.inputs.limit }}"
          fi
          node scripts/semantic/xml-grounded-vertex-rehash.mjs $ARGS

      - name: Upload full-run forensic output
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: semantic-rehash-full
          path: docs/forensics/audits/xml-grounded-vertex-rehash-runs/
---- .github/workflows/deploy-supabase-functions-only.yml ----
name: Deploy Supabase Edge Functions Only

on:
  workflow_dispatch:
  push:
    branches:
      - cleanup/standard-app-tree
    paths:
      - "supabase/migrations/**"
      - "supabase/functions/**"
      - ".github/workflows/deploy-supabase-functions-only.yml"

jobs:
  deploy-functions-and-apply-semantic-schema:
    runs-on: ubuntu-latest
    env:
      SUPABASE_ACCESS_TOKEN: ${{ secrets.SUPABASE_ACCESS_TOKEN }}
      SUPABASE_PROJECT_REF: ${{ secrets.SUPABASE_PROJECT_REF || vars.SUPABASE_PROJECT_REF }}
      MIGRATION_FILE: supabase/migrations/20260617000100_xml_grounded_semantic_rehash.sql

    steps:
      - uses: actions/checkout@v4

      - uses: supabase/setup-cli@v1
        with:
          version: latest

      - name: Validate deploy environment
        run: |
          test -n "$SUPABASE_ACCESS_TOKEN"
          test -n "$SUPABASE_PROJECT_REF"
          test -f "$MIGRATION_FILE"
          supabase --version
          echo "project_ref=$SUPABASE_PROJECT_REF"
          echo "migration_file=$MIGRATION_FILE"
          wc -c "$MIGRATION_FILE"

      - name: Apply XML-grounded semantic schema through Supabase Management API
        run: |
          python - <<'PY'
          import json
          from pathlib import Path

          sql = Path("supabase/migrations/20260617000100_xml_grounded_semantic_rehash.sql").read_text(encoding="utf-8")
          Path("migration-body.json").write_text(
              json.dumps({"query": sql}, ensure_ascii=False),
              encoding="utf-8"
          )
          PY

          curl --fail-with-body --show-error --silent \
            --request POST \
            "https://api.supabase.com/v1/projects/${SUPABASE_PROJECT_REF}/database/query" \
            --header "Authorization: Bearer ${SUPABASE_ACCESS_TOKEN}" \
            --header "Content-Type: application/json" \
            --data @migration-body.json

      - name: Verify semantic schema through Supabase Management API
        run: |
          python - <<'PY'
          import json
          from pathlib import Path

          query = """
          select table_name
          from information_schema.tables
          where table_schema = 'public'
            and table_name in (
              'source_documents',
              'source_document_parts',
              'source_document_revisions',
              'semantic_runs',
              'semantic_batches',
              'archetype_observations',
              'archetype_movements',
              'dualism_relations',
              'dualism_relation_evidence'
            )
          order by table_name;
          """
          Path("verify-body.json").write_text(
              json.dumps({"query": query}, ensure_ascii=False),
              encoding="utf-8"
          )
          PY

          curl --fail-with-body --show-error --silent \
            --request POST \
            "https://api.supabase.com/v1/projects/${SUPABASE_PROJECT_REF}/database/query" \
            --header "Authorization: Bearer ${SUPABASE_ACCESS_TOKEN}" \
            --header "Content-Type: application/json" \
            --data @verify-body.json

      - name: Deploy Supabase Edge Functions
        run: |
          supabase functions deploy create-upload-url --project-ref "$SUPABASE_PROJECT_REF"
          supabase functions deploy register-uploaded-artifact --project-ref "$SUPABASE_PROJECT_REF"
          supabase functions deploy verify-artifact-hash --project-ref "$SUPABASE_PROJECT_REF"
          supabase functions deploy enqueue-enrichment --project-ref "$SUPABASE_PROJECT_REF"
          supabase functions deploy process-enrichment-batch --project-ref "$SUPABASE_PROJECT_REF"
